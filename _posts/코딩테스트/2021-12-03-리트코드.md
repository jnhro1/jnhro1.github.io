```js
var topKFrequent = function (nums, k) {
  const answer = [];
  const obj = nums.reduce((acc, cur) => {
    acc[cur] = (acc[cur] || 0) + 1;
    return acc;
  }, {});
  const sortedArr = Object.entries(obj).sort((x, y) => y[1] - x[1]);
  for (let i = 0; i < k; i++) {
    answer.push(+sortedArr[i][0]);
  }
  return answer;
};

console.log(topKFrequent([1, 1, 1, 2, 2, 3], 2));
```
