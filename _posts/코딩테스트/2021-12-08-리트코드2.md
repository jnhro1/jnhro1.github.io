```js
/**
 * @param {string[]} strs
 * @return {string[][]}
 */
var groupAnagrams = function (strs) {
  let hash = {};
  for (let i = 0; i < strs.length; i++) {
    let sorted = [...strs[i]].sort().join('');
    hash[sorted] ? hash[sorted].push(strs[i]) : (hash[sorted] = [strs[i]]);
  }
  return Object.values(hash);
};
```
