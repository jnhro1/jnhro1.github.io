function gcd(a, b) {
  if (a < b) [a, b] = [b, a];

  while (b > 0) {
    let temp = b;
    b = a % b;
    a = temp;
  }
  return a;
}

function solution(w, h) {
  return w * h - (w + h - gcd(w, h));
}

console.log(solution(8, 12));
